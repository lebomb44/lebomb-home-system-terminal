00064 #define _irq_prolog \
00065     asm volatile ("push r24" "\n\t"                             /* save r24 to current stack */ \
00066                   "push r25" "\n\t"                             /* save r25 to current stack */ \
00067                   "in r24,__SREG__" "\n\t"              /* load SREG into r24 */ \
00068                   "push r24" "\n\t");                   /* and push it to current stack */ 
00069 
00070 #define _irq_epilog \
00071     asm volatile ("pop r24" "\n\t"                              /* restore r24 from stack */ \
00072                   "out __SREG__, r24" "\n\t"    /* write it to SREG */ \
00073                   "pop r25" "\n\t"                              /* load byte from stack to r25 */ \
00074                   "pop r24" "\n\t");                    /* load byte from stack to r24 */
