00176 static void LcdWriteByte(unsigned int data)
00177 {
00178     uint8_t msk = 0x80;
00179 
00180     while (msk) {
00181         LCD_CLK_CLR();
00182         if (data & msk) {
00183             LCD_MOSI_SET();
00184         } else {
00185             LCD_MOSI_CLR();
00186         }
00187         LCD_CLK_SET();
00188         msk >>= 1;
00189         LcdDelay(1);
00190     }
00191     LcdWaitReady(LCD_LONG_DELAY);
00192 }
