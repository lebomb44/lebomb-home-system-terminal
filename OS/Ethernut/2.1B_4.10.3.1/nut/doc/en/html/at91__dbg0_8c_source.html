00068 static int Debug0Init(NUTDEVICE * dev)
00069 {
00070     /* Enable UART clock. */
00071 #if defined (PS_PCER)
00072     outr(PS_PCER, _BV(US0_ID));
00073 #elif defined (PMC_PCER)
00074     outr(PMC_PCER, _BV(US0_ID));
00075 #endif
00076     /* Disable GPIO on UART tx/rx pins. */
00077 #if defined (P15_RXD0) && defined (P14_TXD0)
00078     outr(PIO_PDR, _BV(P15_RXD0) | _BV(P14_TXD0));
00079 #elif defined (PA0_RXD0_A) && defined (PA1_TXD0_A)
00080     outr(PIOA_PDR, _BV(PA0_RXD0_A) | _BV(PA1_TXD0_A));
00081 #elif defined (PA5_RXD0_A) && defined (PA6_TXD0_A)
00082     outr(PIOA_PDR, _BV(PA5_RXD0_A) | _BV(PA6_TXD0_A));
00083 #endif
00084     /* Reset UART. */
00085     outr(US0_CR, US_RSTRX | US_RSTTX | US_RXDIS | US_TXDIS);
00086     /* Disable all UART interrupts. */
00087     outr(US0_IDR, 0xFFFFFFFF);
00088 #if defined (US0_RCR) && defined(US0_TCR)
00089     /* Clear UART counter registers. */
00090     outr(US0_RCR, 0);
00091     outr(US0_TCR, 0);
00092 #endif
00093 #if NUT_DEV_DEBUG_SPEED
00094     /* Set UART baud rate generator register. */
00095     outr(US0_BRGR, At91BaudRateDiv(NUT_DEV_DEBUG_SPEED));
00096 #endif
00097     /* Set UART mode to 8 data bits, no parity and 1 stop bit. */
00098     outr(US0_MR, US_CHMODE_NORMAL | US_CHRL_8 | US_PAR_NO | US_NBSTOP_1);
00099     /* Enable UART receiver and transmitter. */
00100     outr(US0_CR, US_RXEN | US_TXEN);
00101 
00102     return 0;
00103 }
