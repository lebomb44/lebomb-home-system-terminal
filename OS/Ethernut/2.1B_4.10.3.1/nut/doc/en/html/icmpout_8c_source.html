00159 int NutIcmpOutput(uint8_t type, uint32_t dest, NETBUF * nb)
00160 {
00161     ICMPHDR *icp;
00162     uint16_t csum;
00163 
00164     icp = (ICMPHDR *) nb->nb_tp.vp;
00165     icp->icmp_type = type;
00166     icp->icmp_cksum = 0;
00167 
00168     csum = NutIpChkSumPartial(0, nb->nb_tp.vp, nb->nb_tp.sz);
00169     icp->icmp_cksum = NutIpChkSum(csum, nb->nb_ap.vp, nb->nb_ap.sz);
00170 
00171     return NutIpOutput(IPPROTO_ICMP, dest, nb);
00172 }

00189 int NutIcmpReply(uint8_t type, uint8_t code, uint32_t spec, uint32_t dest, NETBUF * nb)
00190 {
00191     ICMPHDR *icp;
00192 
00193     if ((nb = NutNetBufAlloc(nb, NBAF_TRANSPORT, sizeof(ICMPHDR))) == 0)
00194         return -1;
00195 
00196     icp = (ICMPHDR *) nb->nb_tp.vp;
00197     icp->icmp_code = code;
00198     icp->icmp_spec = spec;
00199 
00200     return NutIcmpOutput(type, dest, nb);

00219 int NutIcmpResponse(uint8_t type, uint8_t code, uint32_t spec, NETBUF * nb)
00220 {
00221     IPHDR *ip;
00222     uint32_t dest;
00223 
00224     ip = nb->nb_nw.vp;
00225     dest = ip->ip_src;
00226 
00227     if ((nb = NutNetBufAlloc(nb, NBAF_APPLICATION, sizeof(IPHDR) + 8)) == 0)
00228         return -1;
00229 
00230     memcpy(nb->nb_ap.vp, nb->nb_nw.vp, sizeof(IPHDR));
00231     memcpy((char *)nb->nb_ap.vp + sizeof(IPHDR), nb->nb_tp.vp, 8);
00232 
00233     return NutIcmpReply(type, code, spec, dest, nb);
