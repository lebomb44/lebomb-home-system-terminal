00066 /* Reads line from standard input. */
00067 static int EditLine(char *prompt, char *line, int siz)
00068 {
00069     int ch;
00070     int pos = strlen(line);
00071  
00072     printf("%s: %s", prompt, line);
00073     for (;;) {
00074         ch = getchar();
00075         if (ch == 8) {
00076             if (pos) {
00077                 pos--;
00078                 printf("\b \b");
00079             }
00080         }
00081         else if (ch < ' ') {
00082             break;
00083         }
00084         else if (pos + 1 < siz) {
00085             putchar(ch);
00086             line[pos++] = ch;
00087         }
00088         else {
00089             putchar('\a');
00090         }
00091     }
00092     line[pos] = 0;
00093     putchar('\n');
00094  
00095     return 0;
00096 }
