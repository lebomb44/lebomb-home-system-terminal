00099 void vsyslog_P(int pri, PGM_P fmt_P, va_list ap)
00100 {
00101     /* Build the header. */
00102     size_t cnt = syslog_header(pri);
00103 
00104     if (cnt) {
00105 #ifdef SYSLOG_PERROR_ONLY
00106         fputs(syslog_buf, stderr);
00107         vfprintf_P(stderr, fmt_P, ap);
00108         fputc('\n', stderr);
00109 #else
00110         /* Potentially dangerous. We need vsnprintf() */
00111         if (cnt + strlen_P(fmt_P) >= SYSLOG_MAXBUF) {
00112             return;
00113         }
00114         cnt += vsprintf_P(&syslog_buf[cnt], fmt_P, ap);
00115         syslog_flush(cnt);
00116 #endif /* SYSLOG_PERROR_ONLY */
00117     }
00118 }
