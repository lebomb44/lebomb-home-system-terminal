00176 void NutRegisterTimer(void (*handler) (void *))
00177 {
00178 #if defined(NUT_TICK_AT91PIT)
00179     /* Set compare value for the specified tick frequency. */
00180 #if defined(AT91_PLL_MAINCK)
00181     outr(PIT_MR, (At91GetMasterClock() / (16 * NUT_TICK_FREQ) - 1) << PIT_PIV_LSB);
00182 #else
00183     outr(PIT_MR, (NutGetCpuClock() / (16 * NUT_TICK_FREQ) - 1) << PIT_PIV_LSB);
00184 #endif
00185 
00186     /* Register system interrupt handler. */
00187     NutRegisterSysIrqHandler(&syssig_PIT, handler, NULL);
00188     /* Enable interval timer and interval timer interrupts */
00189     outr(PIT_MR, inr(PIT_MR) | PIT_PITEN | PIT_PITIEN);
00190     NutSysIrqEnable(&syssig_PIT);
00191     inr(PIT_PIVR);
00192 
00193 #else   /* NUT_TICK_AT91PIT */
00194 
00195 #if defined(MCU_AT91SAM7X) || defined(MCU_AT91SAM7S256) || defined(MCU_AT91SAM9260) || defined(MCU_AT91SAM9G45) || defined (MCU_AT91SAM7SE512) || defined(MCU_AT91SAM9XE512)
00196     /* Enable TC0 clock. */
00197     outr(PMC_PCER, _BV(TC0_ID));
00198 #endif
00199 
00200     /* Disable the Clock Counter */
00201     outr(TC0_CCR, TC_CLKDIS);
00202     /* Disable all interrupts */
00203     outr(TC0_IDR, 0xFFFFFFFF);
00204     /* Clear the status register. */
00205     inr(TC0_SR);
00206     /* Select divider and compare trigger */
00207     outr(TC0_CMR, TC_CLKS_MCK32 | TC_CPCTRG);
00208     /* Enable the Clock counter */
00209     outr(TC0_CCR, TC_CLKEN);
00210     /* Validate the RC compare interrupt */
00211     outr(TC0_IER, TC_CPCS);
00212 
00213     /* Register timer interrupt handler. */
00214     NutRegisterIrqHandler(&sig_TC0, handler, 0);
00215     /* Set to lowest priority. */
00216     NutIrqSetPriority(&sig_TC0, 0);
00217 
00218     /* Enable timer 0 interrupts */
00219     NutIrqEnable(&sig_TC0);
00220     //outr(AIC_IECR, _BV(TC0_ID));
00221 
00222     /* Set compare value for 1 ms. */
00223 #if defined(AT91_PLL_MAINCK)
00224     outr(TC0_RC, At91GetMasterClock() / (32 * NUT_TICK_FREQ));
00225 #else
00226     outr(TC0_RC, NutGetCpuClock() / (32 * NUT_TICK_FREQ));
00227 #endif
00228 
00229     /* Software trigger starts the clock. */
00230     outr(TC0_CCR, TC_SWTRG);
00231 
00232 #endif  /* NUT_TICK_AT91PIT */
00233 }

00249 static unsigned int At91GetPllClock(int plla)
00250 {
00251     unsigned int rc;
00252     unsigned int pllr;
00253     unsigned int divider;
00254 
00255     /* 
00256      * The main oscillator clock frequency is specified by the
00257      * configuration. It's usually equal to the on-board crystal.
00258      */
00259     rc = AT91_PLL_MAINCK;
00260 
00261     /* Retrieve the clock generator register of the selected PLL. */
00262 #if defined(CKGR_PLLAR) && defined(CKGR_PLLBR)
00263     pllr = plla ? inr(CKGR_PLLAR) : inr(CKGR_PLLBR);
00264 #else
00265     pllr = inr(CKGR_PLLR);
00266 #endif
00267 
00268     /* Extract the divider value. */
00269     divider = (pllr & CKGR_DIV) >> CKGR_DIV_LSB;
00270 
00271     if (divider) {
00272         rc /= divider;
00273         rc *= ((pllr & CKGR_MUL) >> CKGR_MUL_LSB) + 1;
00274     }
00275     return rc;
00276 }

00283 static uint32_t At91GetProcessorClock(void)
00284 {
00285     unsigned int rc = 0;
00286     unsigned int mckr = inr(PMC_MCKR);
00287 
00288     /* Determine the clock source. */
00289     switch(mckr & PMC_CSS) {
00290     case PMC_CSS_SLOW_CLK:
00291         /* Slow clock selected. */
00292         rc = AT91_SLOW_CLOCK;
00293         break;
00294     case PMC_CSS_MAIN_CLK:
00295         /* Main clock selected. */
00296         rc = AT91_PLL_MAINCK;
00297         break;
00298 #if defined(PMC_CSS_PLLA_CLK)
00299     case PMC_CSS_PLLA_CLK:
00300         /* PLL A clock selected. */
00301         rc = At91GetPllClock(1);
00302         break;
00303 #endif
00304 #if defined(PMC_CSS_PLLB_CLK)
00305     case PMC_CSS_PLLB_CLK:
00306         /* PLL (B) clock selected. */
00307         rc = At91GetPllClock(0);
00308         break;
00309 #elif defined(PMC_CSS_PLL_CLK)
00310     case PMC_CSS_PLL_CLK:
00311         /* PLL (B) clock selected. */
00312         rc = At91GetPllClock(0);
00313         break;
00314 #endif
00315     }
00316 
00317     /* Handle pre-scaling. */
00318     mckr &= PMC_PRES;
00319     mckr >>= PMC_PRES_LSB;
00320     if (mckr < 7) {
00321         rc /= _BV(mckr);
00322     }
00323     else {
00324         rc = 0;
00325     }
00326     return rc;
00327 }

00354 {
00355     uint32_t rc = 0;

00357     if (idx == NUT_HWCLK_CPU) {

00399 uint32_t NutGetTickClock(void)
00400 {
00401     unsigned int rc;
00402 
00403 #if defined(NUT_TICK_AT91PIT)
00404     rc = ((inr(PIT_MR) & PIT_PIV) + 1) * 16;
00405 #else
00406     rc = inr(TC0_RC) * 32;
00407 #endif
00408 
00409     if (rc) {
00410 #if defined(AT91_PLL_MAINCK)
00411         return At91GetMasterClock() / rc;
00412 #else
00413         return NutGetCpuClock() / rc;
00414 #endif
00415     }
00416     return NUT_TICK_FREQ;
00417 }
