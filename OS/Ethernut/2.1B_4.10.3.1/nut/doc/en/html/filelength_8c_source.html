00087 long _filelength(int fd)
00088 {
00089     NUTFILE *fp = (NUTFILE *) ((uintptr_t) fd);
00090     NUTDEVICE *dev;
00091     long l;
00092 
00093     NUTASSERT(fp != NULL);
00094     dev = fp->nf_dev;
00095     if (dev == 0) {
00096         NUTVIRTUALDEVICE *vdv = (NUTVIRTUALDEVICE *) fp;
00097         if (vdv->vdv_ioctl && vdv->vdv_ioctl(vdv, IOCTL_GETFILESIZE, &l) == 0)
00098             return l;
00099         else {
00100             errno = EBADF;
00101             return -1;
00102         }
00103     }
00104 
00105     if (dev->dev_size == 0) {
00106         errno = EBADF;
00107         return -1;
00108     }
00109 
00110     return (*dev->dev_size) (fp);
00111 }
