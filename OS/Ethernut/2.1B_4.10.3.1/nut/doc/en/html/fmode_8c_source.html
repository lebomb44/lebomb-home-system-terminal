00070 int _fmode(CONST char *mode)
00071 {
00072     int mflags;
00073     int pflags;
00074     int bflag = _O_TEXT;
00075 
00076     NUTASSERT(mode != NULL);
00077 
00078     switch (*mode) {
00079     case 'r':
00080         mflags = _O_RDONLY;
00081         pflags = _O_RDWR;
00082         break;
00083     case 'w':
00084         mflags = _O_WRONLY | _O_CREAT | _O_TRUNC;
00085         pflags = _O_RDWR | _O_CREAT | _O_TRUNC;
00086         break;
00087     case 'a':
00088         mflags = _O_WRONLY | _O_CREAT | _O_APPEND;
00089         pflags = _O_RDWR | _O_CREAT | _O_APPEND;
00090         break;
00091     default:
00092         errno = EINVAL;
00093         return EOF;
00094     }
00095     while (*++mode) {
00096         switch (*mode) {
00097         case '+':
00098             mflags = pflags;
00099             break;
00100         case 'b':
00101             bflag = _O_BINARY;
00102             break;
00103         default:
00104             errno = EINVAL;
00105             return EOF;
00106         }
00107     }
00108     return (mflags | bflag);
00109 }
