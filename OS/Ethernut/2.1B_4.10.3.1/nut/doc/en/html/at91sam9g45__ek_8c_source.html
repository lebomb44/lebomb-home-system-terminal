00089 static void Sam9260ekReset(void)
00090 {
00091     /* Set reset pulse length to 250us, disable user reset. */
00092     outr(RSTC_MR, RSTC_KEY | (2 << RSTC_ERSTL_LSB));
00093     /* Invoke external reset. */
00094     outr(RSTC_CR, RSTC_KEY | RSTC_EXTRST);
00095     Sam9g45ekDelay(250);
00096     /* Wait until reset pin is released. */
00097     while ((inr(RSTC_SR) & RSTC_NRSTL) == 0);
00098     Sam9g45ekDelay(25000);
00099 }
