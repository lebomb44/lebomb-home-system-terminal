00168 {
00169     XCPHDR *xch;
00170 
00171     if ((nb = NutNetBufAlloc(nb, NBAF_NETWORK, sizeof(XCPHDR))) == 0)
00172         return -1;
00173 
00174     xch = nb->nb_nw.vp;
00175     xch->xch_code = code;
00176     xch->xch_id = id;
00177     xch->xch_len = htons(nb->nb_nw.sz + nb->nb_tp.sz + nb->nb_ap.sz);
00178 
00179     if (NutPppOutput(dev, PPP_PAP, 0, nb) == 0)
00180         NutNetBufFree(nb);
00181 
00182     return 0;

00188 void PapTxAuthReq(NUTDEVICE *dev, uint8_t id)
00189 {
00190     PPPDCB *dcb = dev->dev_dcb;
00191     NETBUF *nb;
00192     char *cp;
00193     int len;
00194 
00195     /*
00196      * Create the request.
00197      */
00198     len = 2;
00199     if(dcb->dcb_user)
00200         len += strlen((char*)dcb->dcb_user);
00201     if(dcb->dcb_pass)
00202         len += strlen((char*)dcb->dcb_pass);
00203     if ((nb = NutNetBufAlloc(0, NBAF_APPLICATION, len)) != 0) {
00204         cp = nb->nb_ap.vp;
00205         *cp = dcb->dcb_user ? (char)strlen((char*)dcb->dcb_user) : 0;
00206         if(*cp)
00207             memcpy(cp + 1, dcb->dcb_user, *cp);
00208 
00209         cp += *cp + 1;
00210         *cp = dcb->dcb_pass ? (char)strlen((char*)dcb->dcb_pass) : 0;
00211         if(*cp)
00212             memcpy(cp + 1, dcb->dcb_pass, *cp);
00213 
00214         dcb->dcb_auth_state = PAPCS_AUTHREQ;
00215         NutPapOutput(dev, XCP_CONFREQ, ++dcb->dcb_reqid, nb);
00216     }
00217 }
