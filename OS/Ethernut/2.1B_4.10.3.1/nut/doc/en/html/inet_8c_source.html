00131 uint32_t inet_addr(CONST char * str)
00132 {
00133     uint_fast16_t num;
00134     uint32_t addr = 0;
00135     uint_fast8_t parts = 0;
00136     uint8_t *ap;
00137 
00138     ap = (uint8_t *) & addr;
00139     while (parts < 4) {
00140         if (*str < '0' || *str > '9')
00141             break;
00142         for (num = 0; num <= 255;) {
00143             num = (num * 10) + (*str - '0');
00144             if (*++str < '0' || *str > '9')
00145                 break;
00146         }
00147         if (num > 255)
00148             break;
00149         parts++;
00150         *ap++ = (uint8_t) num;
00151         if (*str != '.') {
00152             if (parts == 4)
00153                 return addr;
00154             break;
00155         }
00156         str++;
00157     }
00158     return -1;
00159 }

00175 char *inet_ntoa(uint32_t addr)
00176 {
00177     static char str[16];
00178     char inv[3];
00179     char *rp;
00180     uint8_t *ap;
00181     uint8_t rem;
00182     uint_fast8_t n;
00183     uint_fast8_t i;
00184 
00185     rp = str;
00186     ap = (uint8_t *) & addr;
00187     for (n = 0; n < 4; n++) {
00188         i = 0;
00189         do {
00190             rem = *ap % (uint8_t) 10;
00191             *ap /= (uint8_t) 10;
00192             inv[i++] = '0' + rem;
00193         } while (*ap);
00194         while (i--)
00195             *rp++ = inv[i];
00196         *rp++ = '.';
00197         ap++;
00198     }
00199     *--rp = 0;
00200     return str;
00201 }
