00143 void NutArpInput(NUTDEVICE * dev, NETBUF * nb)
00144 {
00145     ETHERARP *ea;
00146     ARPHDR *ah;
00147     IFNET *nif;
00148 
00149     if (nb->nb_nw.sz < sizeof(ETHERARP)) {
00150         NutNetBufFree(nb);
00151         return;
00152     }
00153 
00154     ea = (ETHERARP *) nb->nb_nw.vp;
00155     ah = (ARPHDR *) & ea->ea_hdr;
00156 
00157     /*
00158      * Silently discard packets with unsupported
00159      * address types and lengths.
00160      */
00161     if (ntohs(ah->ar_hrd) != ARPHRD_ETHER ||
00162         ntohs(ah->ar_pro) != ETHERTYPE_IP ||
00163         ah->ar_hln != 6 || ah->ar_pln != 4) {
00164         NutNetBufFree(nb);
00165         return;
00166     }
00167 
00168     /*
00169      * Silently discard packets for other destinations.
00170      */
00171     nif = dev->dev_icb;
00172     if (ea->arp_tpa != nif->if_local_ip) {
00173         NutNetBufFree(nb);
00174         return;
00175     }
00176 
00177     /*
00178      * TODO: Silently discard packets with our own 
00179      * source address.
00180      */
00181 
00182     /*
00183      * TODO: Discard packets with broadcast source
00184      * address.
00185      */
00186 
00187     /*
00188      * TODO: Discard packets if source IP address
00189      * equals our address. We should send a reply.
00190      */
00191 
00192     /*
00193      * Add the sender to our arp cache. Note, that we don't do
00194      * this with replies only, but also with requests on our
00195      * address. The assumption is that if someone is requesting
00196      * our address, they are probably intending to talk to us,
00197      * so it saves time if we cache their address.
00198      */
00199     NutArpCacheUpdate(dev, ea->arp_spa, ea->arp_sha);
00200 
00201     /*
00202      * Reply to ARP requests.
00203      */
00204     if (htons(ea->ea_hdr.ar_op) == ARPOP_REQUEST) {
00205         NETBUF *nbr =
00206             NutArpAllocNetBuf(ARPOP_REPLY, ea->arp_spa, ea->arp_sha);
00207 
00208         if (nbr) {
00209             if (!NutArpOutput(dev, nbr))
00210                 NutNetBufFree(nbr);
00211         }
00212     }
00213     NutNetBufFree(nb);
00214 }
