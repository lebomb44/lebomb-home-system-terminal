00100 void NutIgmpInput(NUTDEVICE * dev, NETBUF * nb)
00101 {
00102     IGMP *igmp = (IGMP *) nb->nb_tp.vp;
00103     IFNET *nif;
00104     MCASTENTRY *mca;
00105     
00106     nif = dev->dev_icb;    
00107 
00108     /*
00109      * Silently discard packets, which are too small.
00110      */
00111     if (igmp == NULL || nb->nb_tp.sz < IGMP_MINLEN) {
00112         NutNetBufFree(nb);
00113         return;
00114     }
00115 
00116     switch (igmp->igmp_type) {
00117     case IGMP_MEMBERSHIP_QUERY:
00118         /* Clear the received buffer first. */ 
00119         NutNetBufFree(nb);
00120         
00121         /* Go through the list and send reports */
00122         for (mca = nif->if_mcast; mca; mca = mca->mca_next) {
00123             /* Do not send for 224.0.0.1 */ 
00124             if (mca->mca_ip != INADDR_ALLHOSTS_GROUP) {
00125                 NutIgmpJoinGroup(dev, mca->mca_ip);
00126             }
00127         }
00128         break;
00129     case IGMP_V1_MEMBERSHIP_REPORT:
00130     case IGMP_V2_MEMBERSHIP_REPORT:
00131         /* Not supported in the moment */
00132         NutNetBufFree(nb);
00133         break;
00134     default:
00135         /* Unrecognized message types are silently ignored. */
00136         NutNetBufFree(nb);
00137     }
00138 }
