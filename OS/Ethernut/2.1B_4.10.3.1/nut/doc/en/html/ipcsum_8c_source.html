00153 uint16_t NutIpChkSumPartial(uint16_t ics, CONST void *buf, int len)
00154 {
00155     register uint32_t sum = ics;
00156     register uint8_t *cp = (uint8_t *) buf;
00157 
00158     /* Sum up 16 bit values. */
00159     while (len > 1) {
00160 #ifdef __BIG_ENDIAN__
00161         sum += ((uint16_t)*cp << 8) | *(cp + 1);
00162 #else
00163         sum += ((uint16_t)*(cp + 1) << 8) | *cp;
00164 #endif
00165         cp += 2;
00166         len -= 2;
00167     }
00168 
00169     /* Add remaining byte on odd lengths. */
00170     if (len) {
00171 #ifdef __BIG_ENDIAN__
00172         sum += (uint16_t)*cp << 8;
00173 #else
00174         sum += *cp;
00175 #endif
00176     }
00177 
00178     /* Fold upper 16 bits to lower ones. */
00179     while (sum >> 16) {
00180         sum = (uint16_t)sum + (sum >> 16);
00181     }
00182     return (uint16_t) sum;
00183 }
