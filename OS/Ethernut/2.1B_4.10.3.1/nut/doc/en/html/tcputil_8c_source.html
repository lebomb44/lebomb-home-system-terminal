00084  * Calculate round trip time.
00085  */
00086 void NutTcpCalcRtt(TCPSOCKET * sock)
00087 {
00088     uint16_t delta;
00089     
00090     if (sock->so_retran_time == 0)
00091         return;
00092         
00093     delta = (uint16_t) NutGetMillis() - (sock->so_retran_time & ~1);
00094     
00095     /* According to RFC793 (or STD007), page 41, we use 0.8 for ALPHA and 2.0 for BETA. */
00096     sock->so_rtto = min (TCP_RTTO_MAX, max(TCP_RTTO_MIN, (delta * 4 + sock->so_rtto * 8) / 10));

00098 }
