00076 #define IRQ_ENTRY()                                                                                                                                                     \
00077 {                                                                                                                                                                                       \
00078         /* Prevent preempted interrupts disabling global interrupts */                                                  \
00079         __asm__ __volatile__ ("ssrf\t%0; nop; nop" :: "i" (AVR32_SR_GM_OFFSET) : "memory");             \
