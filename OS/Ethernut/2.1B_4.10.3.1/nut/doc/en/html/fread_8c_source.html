00075 size_t fread(void *buffer, size_t size, size_t count, FILE * stream)
00076 {
00077     size_t rc;
00078     size_t nu = 0;
00079 
00080     NUTASSERT(stream != NULL);
00081     NUTASSERT(buffer != NULL);
00082     if (size > 1)
00083         count *= size;
00084     if (stream->iob_flags & _IOUNG) {
00085         stream->iob_flags &= ~_IOUNG;
00086         *((char *) buffer) = (char) stream->iob_unget;
00087         buffer = ((char *) buffer) + 1;
00088         count--;
00089         nu++;
00090     }
00091     rc = (size_t) _read(stream->iob_fd, buffer, count);
00092     if (rc == 0) {
00093         stream->iob_flags |= _IOEOF;
00094     }
00095     else if(rc == (size_t)-1) {
00096         stream->iob_flags |= _IOERR;
00097         rc = 0;
00098     }
00099     rc += nu;
00100     if (size > 1) {
00101         rc /= size;
00102     }
00103     return rc;
00104 }
