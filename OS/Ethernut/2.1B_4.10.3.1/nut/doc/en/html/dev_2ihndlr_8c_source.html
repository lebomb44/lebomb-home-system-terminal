00163 int NutIrqSetPriority(IRQ_HANDLER * irq, int level)
00164 {
00165     int rc = -1;
00166 
00167     if (irq->ir_ctl) {
00168         int prev;
00169 
00170         rc = (irq->ir_ctl) (NUT_IRQCTL_GETPRIO, &prev);
00171         if (rc == 0 && (rc = (irq->ir_ctl) (NUT_IRQCTL_SETPRIO, &level)) == 0) {
00172             rc = prev;
00173         }
00174     }
00175     return rc;
00176 }
