00129 int NutConditionWait(CONDITION * cond)
00130 {
00131     if (cond == NULL) return -1;
00132     NutMutexUnlock(&cond->mutex);
00133     NutEventWait(cond->event, NUT_WAIT_INFINITE);
00134     NutMutexLock(&cond->mutex);
00135     return 0;
00136 }

00166 int NutConditionTimedWait(CONDITION * cond, uint32_t abs_ms)
00167 {
00168     uint32_t ms;
00169 
00170     if (cond == NULL) return -1;
00171     ms = abs_ms - NutGetMillis();
00172     if (ms > 0x7FFFFFFF) return -1;
00173 
00174     NutMutexUnlock(&cond->mutex);
00175     NutEventWait(&cond->event, ms);
00176     NutMutexLock(&cond->mutex);
00177     return 0;
00178 }
