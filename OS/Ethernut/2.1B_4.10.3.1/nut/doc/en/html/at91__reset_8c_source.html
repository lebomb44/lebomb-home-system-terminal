00054 void At91Reset(void)
00055 {
00056 #if defined(RSTC_PROCRST)
00057     /* Use reset controller if available. */
00058     outr(RSTC_CR, RSTC_KEY | RSTC_EXTRST | RSTC_PERRST | RSTC_PROCRST);
00059 #else
00060     /* Otherwise use the watchdog. */
00061     if (At91WatchDogStart(1, 0)) {
00062         for (;;);
00063     }
00064 #endif
00065 }
