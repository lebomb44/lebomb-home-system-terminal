00611         NutTcpAccept(sock, 80);

00642              */
00643             NutHttpProcessRequest(stream);

00646              * Destroy the virtual stream device.
00647              */
00648             fclose(stream);
00649         }
00650

00652          * Close our socket.
00653          */
00654         NutTcpCloseSocket(sock);
