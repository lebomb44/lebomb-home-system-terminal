00099 unsigned long strtoul(CONST char *nptr, char **endptr, int base)
00100 {
00101     register CONST char *s;
00102     register unsigned long acc, cutoff;
00103     register int c;
00104     register int neg, any, cutlim;
00105 
00106     /*
00107      * See strtol for comments as to the logic used.
00108      */
00109     s = nptr;
00110     do {
00111         c = (unsigned char) *s++;
00112     } while (isspace(c));
00113     if (c == '-') {
00114         neg = 1;
00115         c = *s++;
00116     } else {
00117         neg = 0;
00118         if (c == '+')
00119             c = *s++;
00120     }
00121     if ((base == 0 || base == 16) && c == '0' && (*s == 'x' || *s == 'X')) {
00122         c = s[1];
00123         s += 2;
00124         base = 16;
00125     }
00126     if (base == 0)
00127         base = c == '0' ? 8 : 10;
00128 
00129     cutoff = ULONG_MAX / (unsigned long) base;
00130     cutlim = ULONG_MAX % (unsigned long) base;
00131     for (acc = 0, any = 0;; c = (unsigned char) *s++) {
00132         if (isdigit(c))
00133             c -= '0';
00134         else if (isalpha(c))
00135             c -= isupper(c) ? 'A' - 10 : 'a' - 10;
00136         else
00137             break;
00138         if (c >= base)
00139             break;
00140         if (any < 0)
00141             continue;
00142         if ((acc > cutoff || acc == cutoff) && c > cutlim) {
00143             any = -1;
00144             acc = ULONG_MAX;
00145             errno = ERANGE;
00146         } else {
00147             any = 1;
00148             acc *= (unsigned long) base;
00149             acc += c;
00150         }
00151     }
00152     if (neg && any > 0)
00153         acc = -acc;
00154     if (endptr != 0)
00155         *endptr = (char *) (any ? s - 1 : nptr);
00156     return (acc);
00157 }
