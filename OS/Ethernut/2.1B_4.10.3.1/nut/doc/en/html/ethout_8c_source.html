00144 int NutEtherOutput(NUTDEVICE * dev, uint16_t type, uint8_t * ha, NETBUF * nb)
00145 {
00146     ETHERHDR *eh;
00147     IFNET *nif;
00148 
00149     if (NutNetBufAlloc(nb, NBAF_DATALINK, sizeof(ETHERHDR)) == 0)
00150         return -1;
00151 
00152     eh = (ETHERHDR *) nb->nb_dl.vp;
00153     nif = dev->dev_icb;
00154     memcpy(eh->ether_shost, nif->if_mac, 6);
00155     if (ha)
00156         memcpy(eh->ether_dhost, ha, 6);
00157     else
00158         memset(eh->ether_dhost, 0xff, 6);
00159     eh->ether_type = htons(type);
00160 
00161     /*
00162      * Call the network device output routine.
00163      */
00164     if((*nif->if_send) (dev, nb)) {
00165         NutNetBufFree(nb);
00166         return -1;
00167     }
00168     return 0;
00169 }
